# 《微恐咖啡厅》微信小游戏技术实现指南

## 一、微信小游戏平台特性

### 1.1 平台限制与要求
```
包体大小限制:
- 主包: 4MB (可通过分包扩展到8MB)
- 分包: 单个分包2MB，总包8MB
- 资源远程加载: 可通过CDN扩展

性能要求:
- 内存使用: < 200MB (建议<100MB)
- 渲染帧率: 30-60fps
- 首次加载: < 5秒

API限制:
- 文件系统: 50MB本地存储
- 网络请求: 需要配置合法域名
- 广告API: 需接入官方广告组件
```

### 1.2 开发环境搭建
```
推荐技术栈:
- 游戏引擎: Cocos Creator 3.x 或 LayaAir
- 开发语言: TypeScript
- IDE: VS Code + 微信开发者工具

项目结构:
/project
  ├─ assets/           # 资源文件
  ├─ scripts/          # 游戏逻辑
  ├─ scenes/          # 场景文件
  ├─ project.config.json

```

class UIManager {
  private static instance: UIManager;
  private layers: Map<string, cc.Node> = new Map();
  
  // 显示界面
  showPanel(panelName: string, data?: any): Promise<void> {
    return this.loadPanel(panelName).then(panel => {
      panel.active = true;
      panel.getComponent(BasePanel).init(data);
    });
  }
  
  // 加载界面
  private loadPanel(panelName: string): Promise<cc.Node> {
    if (this.layers.has(panelName)) {
      return Promise.resolve(this.layers.get(panelName));
    }
    
    return new Promise((resolve, reject) => {
      cc.resources.load(`ui/${panelName}`, cc.Prefab, (err, prefab) => {
        if (err) {
          reject(err);
          return;
        }
        const panel = cc.instantiate(prefab);
        this.layers.set(panelName, panel);
        cc.director.getScene().addChild(panel);
        resolve(panel);
      });
    });
  }
}

class CoffeeMakingSystem {
  private recipes: Map<string, Recipe> = new Map();
  private currentMaking: MakingProcess | null = null;
  
  // 开始制作
  startMaking(coffeeId: string): boolean {
    const recipe = this.recipes.get(coffeeId);
    if (!recipe || !this.hasEnoughMaterials(recipe)) {
      return false;
    }
    
    this.consumeMaterials(recipe);
    this.currentMaking = {
      coffeeId,
      startTime: Date.now(),
      duration: recipe.duration,
      isAccelerated: false
    };
    
    // 显示制作进度
    this.showProgressUI();
    return true;
  }
  
  // 加速制作（广告）
  accelerateMaking(): void {
    if (!this.currentMaking || this.currentMaking.isAccelerated) {
      return;
    }
    
    // 调用广告API
    wx.createRewardedVideoAd({
      adUnitId: '加速制作广告位'
    }).show().then(() => {
      this.currentMaking.isAccelerated = true;
      this.completeMaking();
    });
  }
  
  // 完成制作
  private completeMaking(): void {
    if (!this.currentMaking) return;
    
    const coffee = this.createCoffee(this.currentMaking.coffeeId);
    this.dispatchEvent('coffee_made', { coffee });
    this.currentMaking = null;
  }
}

// CustomerSystem.ts
class CustomerSystem {
  private customers: Customer[] = [];
  private waitingQueue: Customer[] = [];
  private maxCustomers: number = 5;
  
  // 生成顾客
  generateCustomer(): void {
    if (this.customers.length >= this.maxCustomers) {
      return;
    }
    
    const customerType = this.getRandomCustomerType();
    const customer = new Customer(customerType);
    customer.satisfaction = 100;
    customer.waitTimer = setInterval(() => {
      customer.satisfaction -= 1;
      if (customer.satisfaction <= 0) {
        this.customerLeave(customer, false);
      }
    }, 1000);
    
    this.customers.push(customer);
    this.waitingQueue.push(customer);
    this.updateCustomerUI();
  }
  
  // 服务顾客
  serveCustomer(customerId: string, coffee: Coffee): void {
    const customer = this.customers.find(c => c.id === customerId);
    if (!customer) return;
    
    const satisfactionGain = this.calculateSatisfaction(customer, coffee);
    customer.satisfaction = Math.min(100, customer.satisfaction + satisfactionGain);
    
    // 获得收益
    const reward = this.calculateReward(customer, coffee);
    GameDataManager.instance.addCoins(reward.coins);
    
    // 顾客离开
    setTimeout(() => {
      this.customerLeave(customer, true);
    }, 2000);
  }
  
  // 广告召唤特殊顾客
  summonSpecialCustomer(): void {
    wx.createRewardedVideoAd({
      adUnitId: '召唤顾客广告位'
    }).show().then(() => {
      const specialCustomer = this.createSpecialCustomer();
      this.customers.push(specialCustomer);
      this.updateCustomerUI();
    });
  }
}

// IAPSystem.ts
class IAPSystem {
  private products: Map<string, Product> = new Map();
  
  // 初始化商品
  async initProducts(): Promise<void> {
    try {
      // 获取商品配置
      const res = await wx.request({
        url: 'https://your-server.com/products',
        method: 'GET'
      });
      
      // 向微信注册商品
      const iap = wx.getIAP();
      for (const product of res.data) {
        this.products.set(product.id, product);
        iap.registerProduct({
          productId: product.id,
          productType: 'consumable' // 或 'non_consumable'
        });
      }
    } catch (error) {
      console.error('商品初始化失败:', error);
    }
  }
  
  // 发起购买
  async purchase(productId: string): Promise<boolean> {
    const product = this.products.get(productId);
    if (!product) return false;
    
    try {
      const iap = wx.getIAP();
      const order = await iap.requestPurchase({
        productId,
        // 其他参数...
      });
      
      // 验证订单
      const isValid = await this.verifyOrder(order);
      if (isValid) {
        // 发放商品
        await this.deliverProduct(productId);
        return true;
      }
      return false;
    } catch (error) {
      console.error('购买失败:', error);
      return false;
    }
  }
  
  // 发放商品
  private async deliverProduct(productId: string): Promise<void> {
    const product = this.products.get(productId);
    if (!product) return;
    
    // 根据商品类型发放奖励
    switch (product.type) {
      case 'coins':
        GameDataManager.instance.addCoins(product.amount);
        break;
      case 'diamonds':
        GameDataManager.instance.addDiamonds(product.amount);
        break;
      case 'item':
        InventoryManager.instance.addItem(product.itemId, product.amount);
        break;
    }
    
    // 记录购买
    Analytics.trackPurchase(productId);
  }
}

// 1. 纹理内存管理
class TextureManager {
  private static cache: Map<string, cc.Texture2D> = new Map();
  
  static async loadTexture(path: string): Promise<cc.Texture2D> {
    if (this.cache.has(path)) {
      return this.cache.get(path);
    }
    
    return new Promise((resolve, reject) => {
      cc.resources.load(path, cc.Texture2D, (err, texture) => {
        if (err) {
          reject(err);
        } else {
          this.cache.set(path, texture);
          resolve(texture);
        }
      });
    });
  }
  
  static releaseTexture(path: string): void {
    if (this.cache.has(path)) {
      const texture = this.cache.get(path);
      texture.destroy();
      this.cache.delete(path);
    }
  }
}

// 2. 对象池管理
class ObjectPool<T> {
  private pool: T[] = [];
  private createFn: () => T;
  private resetFn: (obj: T) => void;
  
  get(): T {
    if (this.pool.length > 0) {
      return this.pool.pop();
    }
    return this.createFn();
  }
  
  release(obj: T): void {
    this.resetFn(obj);
    this.pool.push(obj);
  }
}

// 1. 请求合并
class RequestBatcher {
  private batch: Map<string, any[]> = new Map();
  private timer: number | null = null;
  
  addRequest(key: string, data: any): void {
    if (!this.batch.has(key)) {
      this.batch.set(key, []);
    }
    this.batch.get(key).push(data);
    
    // 延迟发送
    if (!this.timer) {
      this.timer = setTimeout(() => this.flush(), 1000);
    }
  }
  
  private flush(): void {
    for (const [key, batchData] of this.batch) {
      this.sendBatch(key, batchData);
    }
    this.batch.clear();
    this.timer = null;
  }
}

// 2. 数据压缩
class DataCompressor {

    // 可考虑使用pako等库进一步压缩

  }

// 3. 缓存策略
class NetworkCache {
  private static cache: Map<string, {data: any, timestamp: number}> = new Map();
  private static TTL = 5 * 60 * 1000; // 5分钟
  
  static get(key: string): any | null {
    const item = this.cache.get(key);
    if (!item) return null;
    
    if (Date.now() - item.timestamp > this.TTL) {
      this.cache.delete(key);
      return null;
    }
    
    return item.data;
  }
  
  static set(key: string, data: any): void {
    this.cache.set(key, {
      data,
      timestamp: Date.now()
    });
  }
}

// prefabs/UIButton.ts
const { ccclass, property } = cc._decorator;

@ccclass
export default class UIButton extends cc.Component {
  @property(cc.SpriteFrame)
  normalSprite: cc.SpriteFrame = null;
  
  @property(cc.SpriteFrame)
  pressedSprite: cc.SpriteFrame = null;
  
  @property(cc.SpriteFrame)
  disabledSprite: cc.SpriteFrame = null;
  
  @property(cc.AudioClip)
  clickSound: cc.AudioClip = null;
  
  private sprite: cc.Sprite = null;
  private isEnabled: boolean = true;
  
  onLoad(): void {
    this.sprite = this.node.getComponent(cc.Sprite);
    this.node.on(cc.Node.EventType.TOUCH_START, this.onTouchStart, this);
    this.node.on(cc.Node.EventType.TOUCH_END, this.onTouchEnd, this);
    this.node.on(cc.Node.EventType.TOUCH_CANCEL, this.onTouchCancel, this);
  }
  
  private onTouchStart(): void {
    if (!this.isEnabled) return;
    this.sprite.spriteFrame = this.pressedSprite;
    this.node.scale = 0.95;
  }
  
  private onTouchEnd(): void {
    if (!this.isEnabled) return;
    this.sprite.spriteFrame = this.normalSprite;
    this.node.scale = 1;
    
    // 播放音效
    if (this.clickSound) {
      cc.audioEngine.playEffect(this.clickSound, false);
    }
    
    // 触发点击事件
    this.node.emit('click');
  }
  
  setEnabled(enabled: boolean): void {
    this.isEnabled = enabled;
    this.sprite.spriteFrame = enabled ? this.normalSprite : this.disabledSprite;
  }
}
