```typescript
// 核心实现原理
class TokenMonitor {
  private currentTokenCount: number = 0;
  private readonly maxTokens: number = 115000; // 预留5k安全空间
  private stateFile: string = "./project_state.md";
  
  // 实时监控token使用
  monitorConversation(userInput: string, aiResponse: string): void {
    const inputTokens = this.estimateTokens(userInput);
    const outputTokens = this.estimateTokens(aiResponse);
    this.currentTokenCount += inputTokens + outputTokens;
    
    // 达到阈值时自动触发断点保存
    if (this.currentTokenCount >= this.maxTokens) {
      this.saveProjectState();
      this.triggerAutoReset();
    }
  }
  
  // 自动保存项目状态
  private saveProjectState(): void {
    // 压缩状态，最小化token占用
    const compressedState = this.compressProjectState();
    fs.writeFileSync(this.stateFile, compressedState);
  }
}
```
